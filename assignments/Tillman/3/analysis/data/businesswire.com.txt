	$(function () {
		var expiredate = new Date('2021-08-17T00:00:00');
		var now = new Date();
		var ieExpired = now.getTime() > expiredate.getTime();
		//if we find var from config, use it
		if (typeof isIEExpired !== 'undefined') {
			ieExpired = isIEExpired;
		}
		//Name cookie
		var cookieName = 'IEWarningCookie';
		//Read cookie
		var cookieFound = getCookie(cookieName);
		// Microsoft Internet Explorer detected in.
		if (!cookieFound
			&& isIE()) {
			showIEWarning(true);
		} else {
			showIEWarning(false);
		}

		$('#closeButton').on('click', function () {
			setCookie(cookieName, 'true');
			showIEWarning(false);
		});

		function showIEWarning(show) {
			if (show) {
				if ($('#bw-home').length) {
					$('#bw-home').addClass('ie-warn');
				}
				if ($('#bw-main').length) {
					$('#bw-main').addClass('ie-warn');
				}
				if($('#ie-warn').length) {
					$('#ie-warn').css('display', 'block');
					$('.ieWarningSpan').css('display', 'block');
					$('#closeButton').css('display', 'block');
					if (ieExpired) {
						$('#expireContent').css('display', 'block');
					} else {
						$('#warningContent').css('display', 'block');
					}
				}
			} else {
				if ($('#bw-home').length) {
					$('#bw-home').removeClass('ie-warn');
				}
				if ($('#bw-main').length) {
					$('#bw-main').removeClass('ie-warn');
				}
				if($('#ie-warn').length) {
					$('#ie-warn').css('display', 'none');
					$('.ieWarningSpan').css('display', 'none');
					$('#closeButton').css('display', 'none');
					if (ieExpired) {
						$('#warningContent').css('display', 'none');
					} else {
						$('#warningContent').css('display', 'none');
					}
				}
			}
		}

		// Create cookie
		function setCookie(key, value) {
			document.cookie = key + "=" + value + "; expires=30; path=/;secure=true;domain=" + '.businesswire.com';
		}

		function getCookie(key) {
			var allCookies = document.cookie;
			var thisCookiePos = allCookies.indexOf(key);
			if (thisCookiePos != -1) {
				return true;
			}
			return false;
		}

		function isIE() {
			if (navigator && navigator.userAgent) {
				var ua = navigator.userAgent.toUpperCase();
				if (ua.indexOf("MSIE ") > -1
					|| ua.indexOf("TRIDENT/") > -1) {
					return true;
				}
			}
			return false;
		}

		$('#login_form').submit(function() {
			if(isIE() && ieExpired ) {
				if ($('#ie-expire').length) {
					$('#ie-expire').css("display","block");
					return false;
				}
			} else {
				return true;
			}
		});

		$('#closeIEExpireIcon').on('click', function() {
			$('#ie-expire').css('display','none');
		});
	});
