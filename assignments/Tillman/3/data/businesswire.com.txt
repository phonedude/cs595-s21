				if ($('#bw-home').length) {
					$('#bw-home').addClass('ie-warn');
				}
				if ($('#bw-main').length) {
					$('#bw-main').addClass('ie-warn');

						$('#expireContent').css('display', 'block');
					} else {
						$('#warningContent').css('display', 'block');
					}
				}
			} else {
				if ($('#bw-home').length) {
					$('#bw-home').removeClass('ie-warn');
				}
				if ($('#bw-main').length) {
					$('#bw-main').removeClass('ie-warn');

					} else {
						$('#warningContent').css('display', 'none');
					}
				}
			}
		}

		// Create cookie
		function setCookie(key, value) {
			document.cookie = key + "=" + value + "; expires=30; path=/;secure=true;domain=" + '.businesswire.com';
		}

		function getCookie(key) {
			var allCookies = document.cookie;
			var thisCookiePos = allCookies.indexOf(key);
			if (thisCookiePos != -1) {
				return true;
			}
			return false;
		}

		function isIE() {
			if (navigator && navigator.userAgent) {
				var ua = navigator.userAgent.toUpperCase();
				if (ua.indexOf("MSIE ") > -1
					|| ua.indexOf("TRIDENT/") > -1) {
					return true;
				}
			}
			return false;
