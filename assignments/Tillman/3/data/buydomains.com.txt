    // check the MakeOffer form from submit click, if the user fields are all valid, continue with the captcha execute()
    function checkMakeOfferForm() {
    
        if (typeof event !== 'undefined') event.preventDefault();

        // check phone for repeating digits (spammers put them in)        
        document.getElementById('err_phonefield_repeating').style.display="none";
        var checkPhoneRepeats = /(\d)\1{5}/;
        var inValidPhone = checkPhoneRepeats.exec(angular.element(document.getElementById('priContactForm')).scope().visitor.phone);
        if (inValidPhone) {
            document.getElementById('err_phonefield_repeating').style.display="block";
            return false;
        }

        // form isn't valid, apply scope to show errors for user to finish
        if (!angular.element(document.getElementById('priContactForm')).scope().priContactForm.$valid) {
            angular.element(document.getElementById('priContactFormControllerContainer')).scope().submit(angular.element(document.getElementById('priContactForm')).scope().priContactForm.$valid);
            angular.element(document.getElementById('priContactFormControllerContainer')).scope().$apply(); 
            return false;
        }

                
        // if we have a valid form, go ahead and do captcha verify
        grecaptcha.execute();
    }

    // captcha has verified the user is valid, submit form, capture the response key to confirm submission on the server
    function submitMakeOffer() {
        window.grecaptchaValue = grecaptcha.getResponse();
